### 1. HelloWorld

```c
/*
    IDE 통합개발환경
    (Integrated Development Environment)
    Text Editor
    Debug
    Profiler
    Package
    Build - Compile
    Release

    C Language
    Low - Level Language
    High - Level Language
    Machine Language
    Assembly Language
    Compile
    Code - Coding

    절차 '지향'
    (Procedural 'Oriented' Programming)
*/

// 주석(Comments)
/*
    여기는
    전부
    주석입니다.
 */
 // #: 전처리기(Pre-Processor)
 // include: 포함시키다
 // <>: 경로
 // stdio: Standard Input/Ouput
 // .h: Header File, Declaration
 // .c: Source File, Definition
#include <stdio.h>

// 함수(Function)
// 반환형 함수명(매개변수)
// 자료형(Data-Type)
// int: Integer(정수)
// main: 함수명, Entry Point, Entry Function
// (): 매개변수(Parameter)
int main(void)
{
    // 함수호출(Function Call)
    // printf: Print Format
    // 문자(Character): ''
    // 문자열(String): ""
    // \n: 개행문자
    // 인자(Arguments)
    printf("Hello, World!\n");

    // 함수종료, 결과값을 반환
    // Keyword
    // Error-Code: -1
    return 0;
}
```
