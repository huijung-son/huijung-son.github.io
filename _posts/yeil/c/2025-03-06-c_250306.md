### 1. Operators

```c
#include <stdio.h>


void PrintQuest(unsigned char _quest);

int main()
{
	// 연산자(Operators)
	// = : 대입연산자(Assignment Operator)
	int val = 0;


	// 사칙연산
	// 산술연산자(Arithmetic Operator)
	int lhs = 7;	// Left-Hand Side
	int rhs = 3;	// Right-Hand Side

	printf("%d + %d = %d\n", lhs, rhs, lhs + rhs);
	printf("%d - %d = %d\n", lhs, rhs, lhs - rhs);
	printf("%d * %d = %d\n", lhs, rhs, lhs * rhs);

	// 형변환(Type Casting, Type Conversion)
	// 묵시적 형변환
	float fLhs = lhs;
	printf("fLhs: %f\n", fLhs);
	fLhs = 3.14f;
	// 명시적 형변환
	int iLhs = (int)fLhs;
	printf("iLhs: %d\n", iLhs);

	printf("%d / %d = %.2f\n",
			// 자료형의 승격
			lhs, rhs, (float)lhs / rhs);
	// 나머지 연산자
	printf("%d %% %d = %d\n",
			lhs, rhs, lhs % rhs);

	printf("Result: %d\n",
		    (2 - 5) / 3 * (2 + 8));


	// 복합대입연산자
	lhs = 0;
	rhs = 1;

	lhs += rhs;	// lhs = lhs + rhs;
	printf("lhs += rhs: %d\n", lhs);
	rhs -= 2;
	printf("rhs -= 2: %d\n", rhs);
	lhs *= rhs;
	printf("lhs *= rhs: %d\n", lhs);
	lhs /= rhs;
	printf("lhs /= rhs: %d\n", lhs);
	rhs %= 2;
	printf("rhs %= 2: %d\n", rhs);


	printf("\n");


	// 단항연산자(Unary Operator)
	lhs = 0;
	rhs = 0;

	printf("++lhs: %d\n", ++lhs);	// Pre-Increment
	printf("lhs++: %d\n", lhs++);	// Post-Increment
	printf("lhs: %d\n", lhs);

	printf("--rhs: %d\n", --rhs);	// Pre-Decrement
	printf("rhs--: %d\n", rhs--);	// Post-Decrement
	printf("rhs: %d\n", rhs);


	printf("\n");


	// 관계연산자(Relational Operator)
	// 비교연산자, Condition
	lhs = 1;	// TRUE
	rhs = 0;	// FALSE

	printf("lhs > rhs: %d\n", lhs > rhs);
	printf("lhs >= rhs: %d\n", lhs >= rhs);
	printf("lhs < rhs: %d\n", lhs < rhs);
	printf("lhs <= rhs: %d\n", lhs <= rhs);
	printf("lhs == rhs: %d\n", lhs == rhs);
	printf("lhs != rhs: %d\n", lhs != rhs);

	// 부동소수점 오차
	float fVal = 0.0f;
	for (int i = 0; i < 1000; ++i)
		fVal += 0.001f;
	printf("fVal: %f\n", fVal);
	printf("fVal == 1.0f: %d\n", fVal == 1.0f);


	printf("\n");


	// 논리연산자(Logical Operator)
	printf("lhs && rhs: %d\n", lhs && rhs);	// AND
	printf("lhs || rhs: %d\n", lhs || rhs);	// OR
	printf("!lhs: %d\n", !lhs);	// NOT
	printf("7 && 10: %d\n", 7 && 10);
	printf("%d\n",
			lhs && ((rhs || !rhs) && lhs) && rhs);

	int 한글변수 = 0;

	int result = 1 || (lhs = 10);
	printf("lhs: %d\n", lhs);


	printf("\n");


	// 삼항연산자(Ternary or Conditional Operator)
	// 비교문 ? 참 : 거짓
	printf("%d > %d: %c\n",
			lhs, rhs, lhs > rhs ? 'T' : 'F');


	printf("\n");


	// 비트연산자(Bitwise Operator)
	lhs = 5;
	rhs = 3;

	//   00000101
	// & 00000011
	//   00000001
	printf("lhs & rhs: %d\n", lhs & rhs);	// AND
	//   00000101
	// | 00000011
	//   00000111
	printf("lhs | rhs: %d\n", lhs | rhs);	// OR
	//   00000101
	// ^ 00000011
	//   00000110
	printf("lhs ^ rhs: %d\n", lhs ^ rhs);	// XOR
	//   00000101
	// ~ 11111010
	printf("~lhs: %d\n", ~lhs);				// NOT
	
	lhs &= rhs;
	lhs |= rhs;
	lhs ^= rhs;
	lhs = ~lhs;


	printf("\n");


	lhs = 1;

	// Left-Shift
	// 00001101	== 1+4+8 = 13
	// 00011010 == 2+8+16 = 26
	lhs = lhs << 4;
	printf("lhs: %d\n", lhs);
	// Right-Shift
	lhs >>= 2;
	printf("lhs: %d\n", lhs);
		

	printf("\n\n");


	//////////////////////////////////////////


	// Bit-Flags
	unsigned char quest = 0;

	//quest = 4;
	quest |= 1 << (3 - 1);
	PrintQuest(quest);

	//quest += 64;
	quest |= 1 << (7 - 1);
	PrintQuest(quest);

	printf("Q7: %c\n", (quest & 64) > 0 ? 'C' : 'N');

	quest ^= 1 << (7 - 1);
	PrintQuest(quest);


	quest = 0;
	int q2 = 1 << (2 - 1);
	int q5 = 1 << (5 - 1);
	int q8 = 1 << (8 - 1);
	quest = q2 | q5 | q8;
	PrintQuest(quest);


	return 0;
}

void PrintQuest(unsigned char _quest)
{
	for (int i = 0; i < 8; ++i)
	{
		printf("[%c]", (_quest & 128) > 0 ? 'C' : 'N');
		//printf("%d", (_quest & 128) > 0 ? 1 : 0);
		_quest <<= 1;
	}
	printf("\n");
}
```
