```c#
namespace KCH
{
    struct SPoint
    {
        // 멤버 접근 지정자(Member Access Modifier)
        // 필드(Fields)
        // 정보은닉(Information Hiding)
        private int x;
        private int y;

        public static int staticVal = 0;


        // 기본 생성자(Default Constructor)
        public SPoint()
        {
            Console.WriteLine("SPoint Constructor Call!");

            x = 3;
            y = 3;
        }

        // 오버로딩 생성자
        // (Overloading Constructor)
        public SPoint(int _x, int _y)
        {
            Console.WriteLine(
                "SPoint Overloading Constructor Call!");

            x = _x;
            y = _y;
        }

        public SPoint(SPoint _pt)
        {
            Console.WriteLine(
                "SPoint Copy Constructor Call!");

            x = _pt.x;
            y = _pt.y;
        }

        public void Print()
        {
            staticVal = 50;

            Console.WriteLine("(" + x + ", " + y + ")");
        }

        static public void Print(int _x, int _y)
        {
            staticVal = 100;
            //x = 10;

            Console.WriteLine(
                "Overload: (" + _x + ", " + _y + ")");
        }

        //public void AddedValue()
        //{
        //    static int val = 0;
        //    val += 5;
        //}

        // 연산자 오버로딩(Operator Overloading)
        public static SPoint operator +(
            SPoint _pt1, SPoint _pt2)
        {
            return new SPoint(
                _pt1.x + _pt2.x,
                _pt1.y + _pt2.y);
        }
    }
}
```

```c#
class Vector
{
    #region Fields
    private float x = 0.0f; // Member Initializer
    private float y = 0f;
    private float z = .5f;

    public float w { get; set; }

    private const int constVal = 10;
    private readonly int readonlyVal = 9;

    //private const Vector cVec = new Vector();
    //private readonly Vector rVec = new Vector();
    #endregion

    // Properties
    public float Y { get; set; }
    public float Z
    {
        get { return z; }
        set {
            if (value < 0f) value = 0f;
            z = value;
        }
    }


    public Vector()
    {
        Console.WriteLine(
            "Vector Default Constructor Call!");

        this.x = 0f;
        this.y = 0f;
        this.z = 0f;

        //constVal = 7;
        readonlyVal = 1;
        readonlyVal = 100;
    }

    // 소멸자(Destructor)
    ~Vector()
    {
        Console.WriteLine("Vector Destructor Call!");
    }

    public void Foo()
    {
        //readonlyVal = 10;
        //readonly int val = 0;
        const int val = 0;
    }

    // Getter, Setter
    public float GetX()
    {
        return x;
    }

    public void SetX(float x)
    {
        if (x < 0f) x = 0f;
        this.x = x;
    }
}
```
