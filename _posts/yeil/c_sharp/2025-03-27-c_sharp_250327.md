```c#
// 캡슐화(Encapsulation)

class Character
{
    public enum EJob { WitchDoc, Baba, Bard, None }

    private EJob job = EJob.None;
    private sbyte lv = 0;
    private float hp = 0f;
    private float mp = 0f;
    private float atk = 0f;
    private float def = 0f;

    private uint exp = 0;

    // IS-A
    // HAS-A
    private Weapon? weapon = null;


    public float Atk {
        set { atk = value; }
        get { return atk; }
    }


    public Character(
        EJob _job,
        sbyte _lv,
        float _hp, float _mp,
        float _atk, float _def)
    {
        job = _job;
        lv = _lv;
        hp = _hp;
        mp = _mp;
        atk = _atk;
        def = _def;
    }

    public void Movement()
    {
        Console.WriteLine($"{job} is moving");
    }

    public void Attack()
    {
        Console.WriteLine(
            $"{job} is attack: {atk}" +
            (weapon != null ? $" + {weapon.Atk}" : ""));
    }

    public virtual void Damage(float _dmg)
    {
        Console.WriteLine($"{job}: Damage({_dmg})");
        hp -= _dmg;
        if (hp < 0f)
            Console.WriteLine($"{job} is dead");
    }

    public void SetWeapon(Weapon _weapon)
    {
        weapon = _weapon;
    }
}
```

```c#
abstract class Weapon
{
    public enum EElement { Normal, Fire, Ice, Rock }

    private EElement element = EElement.Normal;
    
    private float atk = 0f;


    public float Atk { get { return atk; } }


    public Weapon(float _atk)
    {
        atk = _atk;
    }

    public void SetElement(EElement _element)
    {
        element = _element;
    }

    public abstract void Skill();
}
```

### 2. Collections

```c#
using System.Collections;

class Program
{
    private static void ArrayListExample()
    {
        //System.Array
        System.Collections.ArrayList arrList =
            new System.Collections.ArrayList();

        // Boxing / Unboxing
        arrList.Add(1);
        arrList.Add(3.14);
        arrList.Add("Hello");

        //arrList.Sort();

        Console.Write("ArrayList: ");
        foreach (object obj in arrList)
            Console.Write(obj + " - ");
        Console.WriteLine("({0})", arrList.Count);
    }

    private static void StackExample()
    {
        Stack stack = new Stack();

        stack.Push(1);
        stack.Push("World");

        int cnt = stack.Count;
        for (int i = 0; i < cnt; ++i)
            Console.WriteLine("Stack: " + stack.Pop());
    }

    private static void QueueExample()
    {
        Queue queue = new Queue();

        queue.Enqueue("QUEUE");
        queue.Enqueue(123);

        int cnt = queue.Count;
        for (int i = 0; i < cnt; ++i)
            Console.WriteLine(
                "Queue: " + queue.Dequeue());
    }

    private static void HashtableExample()
    {
        Hashtable ht = new Hashtable();

        // Key, Value
        ht.Add('A', 100);
        ht.Add(7, "KCH");

        Console.WriteLine("ht['A']: " + ht['A']);
        Console.WriteLine("ht[7]: " + ht[7]);

        var i = 1;
        var c = 'a';
        var str = "Hello";
        //var val;
        foreach (var o in ht.Keys)
        {
            Console.WriteLine($"ht[{o}]: {ht[o]}");
        }
    }


    private static void Main()
    {
        // Collections
        ArrayListExample();
        Console.WriteLine();
        StackExample();
        Console.WriteLine();
        QueueExample();
        Console.WriteLine();
        HashtableExample();
    }
}
```

### summary

- 캡슐화(Encapsulation)
- 솔리드원칙(SOLID)
  - SRP(Single Responsibility Principle) 단일 책임 원칙
  - OCP(Open Closed Principle) 개방 폐쇄 원칙
  - LSP(Liskov Substitution Principle) 리스코프 치환 원칙
  - ISP(Interface Segregation Principle) 인터페이스 분리 원칙
  - DIP(Dependency Inversion Principle) 의존 역전 원칙
- IS-A: ~ 는 ~ 이다
- HAS-A: ~ 는 ~ 를 가지고 있다
- 의존성 주입 \| 결합도와 응집도
  - 적절한 의존성과 낮은 결합도 높은 응집도를 생각한다면 좋은 모듈화가 될 수 있다, 솔리드 원칙 참고
- Boxing, UnBoxing
  - 오버헤드와 가비지를 생성 최적화 면에서 최악 
  - Console.WriteLine 함수가 매개변수로 object 로 받으면 되지만 그렇게 안하는 이유를 생각해보자
  - foreach 코드에서 발생하기 쉬움
- Hashtable
  - key, value 모두 object 타입으로 받을 수 있다 그러므로 박싱 언박싱이 일어난다
  - 처리 기준에서 key 는 문자보다 숫자가 읽는 속도가 더 빠르다
